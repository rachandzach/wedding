        // This script makes the 'Add to Calendar' buttons functional.

        document.addEventListener('DOMContentLoaded', function() {

            // Select all buttons that are intended to create a calendar event.

            // We identify them by the 'data-event-title' attribute.

            const calendarButtons = document.querySelectorAll('button[data-event-title]');


            calendarButtons.forEach(button => {

                // Add a click event listener to each button.

                button.addEventListener('click', function() {

                    // Retrieve the event details from the button's data attributes.

                    // encodeURIComponent ensures special characters in titles, etc., are handled correctly in the URL.

                    const title = encodeURIComponent(this.dataset.eventTitle);

                    const start = this.dataset.eventStartUtc; // The start time in UTC format (e.g., 20250719T010000Z)

                    const end = this.dataset.eventEndUtc;     // The end time in UTC format

                    const location = encodeURIComponent(this.dataset.eventLocation);

                    const description = encodeURIComponent(this.dataset.eventDescription);


                    // Construct the Google Calendar URL with the event details.

                    // Using UTC times with the 'Z' suffix ensures Google Calendar shows the event

                    // in the user's local timezone correctly.

                    const googleCalendarUrl = `https://calendar.google.com/calendar/render?action=TEMPLATE&text=${title}&dates=${start}/${end}&details=${description}&location=${location}&sf=true&output=xml`;


                    // Open the Google Calendar link in a new tab.

                    window.open(googleCalendarUrl, '_blank');

                });

            });

        });
